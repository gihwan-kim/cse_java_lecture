public class good {

	private static int x;

	public static void main(String[] args) {
		// 위 내용이 반드시 있어야 함.
		
		// method(함수 라고도 함) 여러개 할 수 있다. 밑의 것
		// ; 문장이 끝났음을 표시 온점 \과 같은 의미
		// System.out.println : 화면에 구현해 달라는 의미.
		// ln : 엔터 역할

		System.out.println("나도 프로그래밍 할수 ㅣㅆ따아아아아아아아어날너라널나ㅣㅓㄹ니ㅏ러내ㅑ러내랴넝래냐ㅓㄹ");
		// 자바에서 변수를 사용하기 전에 사용한다고 언급을 해야 한다.
		int x; // 정수를 넣을 메모리공간을 확보라고 그이름을 x라 한다. 메모리에 값을 넣으면 뺄수도 있겠지.
		// 정수만 넣을 수 있는 공간에는 정수만 넣어야 한다. 그렇지 않을 경우 자바가 알아서 바꾼다. 구분해서 할것
	
		x =20; // = 은 수학에서의 equal의 의미가 아니다. " ' =  ' 의 오른쪽에 있는 값을 저장하라는 의미이다. => 20을 x에 저장하라"
		//오류 표시를 항상 주의 하면서 할것!
		// "ctrl + / "를 같이 누르면 주석을 달게 하거나 없애게 할 수 있다. 
		System.out.println("나의 나이는");
		System.out.println(x);// 변수 x가 가지고 있는 값을 출력 한다.
		System.out.println("샬 입니다아아앙.");
	}

}
